@PostMapping("/api/naac/universities")
	public ResponseEntity<University> saveUniversity(@Valid @RequestBody University university) {
		University savedUniversity = universityRepository.save(university);

		URI location = ServletUriComponentsBuilder.fromCurrentRequest()
						.path("/{id}")
						.buildAndExpand(savedUniversity.getId())
						.toUri();

		return ResponseEntity.created(location).body(savedUniversity);
	}


	@PostMapping("/api/naac/basic-info")
	public ResponseEntity<University> createUniversity(@Valid @RequestBody University university) {
		
		University savedUniversity = universityRepository.save(university);

		URI location = ServletUriComponentsBuilder.fromCurrentRequest()
						.path("/{id}")
						.buildAndExpand(savedUniversity.getId())
						.toUri();   
		
		return ResponseEntity.created(location).build();
